18  package de.aikiit.spamprotector.converter;
19  
20  import org.junit.Test;
21  
22  import static org.junit.Assert.assertEquals;
23  import static org.junit.Assert.assertNull;
24  
25  /**
26   * @author hirsch
27   * @version 2015-12-05, 21:53
28   */
29  public class SpamProtectorTest {
30  
31      @Test
32      public void conversionRoundtripNull() {
33          assertNull(SpamProtector.toEncoded(null));
34          assertNull(SpamProtector.toPlain(null));
35  
36          assertEquals("", SpamProtector.toEncoded(""));
37          assertEquals("", SpamProtector.toPlain(""));
38      }
39  
40      @Test
41      public void conversionExample() {
42          assertEquals(CharacterConverter.SPACE.getReplacement(), SpamProtector.toEncoded(" "));
43          assertEquals(CharacterConverter.SPACE.getPlain(), SpamProtector.toPlain("&nbsp;"));
44      }
45  
46      @Test
47      public void singleCharactersAreConverted() {
48          assertEquals("&nbsp;", SpamProtector.toEncoded(String.valueOf(' ')));
49          assertEquals("&nbsp;", SpamProtector.toEncoded(" "));
50      }
51  
52      @Test
53      public void asciiOnlyIsEncodedCompletelyWithSpacesAndAtSign() {
54          String input = "mailto:w oo@suck.er";
55          assertEquals("&#109;&#97;&#105;&#108;&#116;&#111;&#58;&#119;&nbsp;&#111;" +
56                  "&#111;&#64;&#115;&#117;&#99;&#107;&#46;&#101;&#114;", SpamProtector.toEncoded(input));
57      }
58  
59      @Test
60      public void nonAsciCharactersAreNotEncodedButTheRest() {
61          String input = "mailto:wフィリップoo@suck.нетer";
62          assertEquals("&#109;&#97;&#105;&#108;&#116;&#111;&#58;&#119;フィリップ&#111;" +
63                  "&#111;&#64;&#115;&#117;&#99;&#107;&#46;нет&#101;&#114;", SpamProtector.toEncoded(input));
64      }
65  
66      @Test
67      public void asciiIsConvertedToPlainText() {
68          String input = "mailto:w oo@suck.er";
69          String encoded = "&#109;&#97;&#105;&#108;&#116;&#111;&#58;&#119;&nbsp;&#111;&#111;&#64;&#115;&#117;&#99;&#107;&#46;&#101;&#114;";
70  
71          assertEquals(input, SpamProtector.toPlain(encoded));
72      }
73  
74      @Test
75      public void mixOfEncodedAndNonAsciiPlainCanBeDecoded() {
76          String plain = "mailto:wフィリップoo@suck.нетer";
77          String encoded = "&#109;&#97;&#105;&#108;&#116;&#111;&#58;&#119;フィリップ&#111;&#111;&#64;&#115;&#117;&#99;&#107;&#46;нет&#101;&#114;";
78          assertEquals(plain, SpamProtector.toPlain(encoded));
79      }
80  
81      @Test
82      public void stupidInputDecode() {
83          assertEquals("", SpamProtector.toPlain(""));
84          assertNull(SpamProtector.toPlain(null));
85          assertEquals("²", SpamProtector.toPlain("²"));
86      }
87  
88      @Test
89      public void stupidInputEncode() {
90          assertEquals("", SpamProtector.toEncoded(""));
91          assertNull(SpamProtector.toEncoded(null));
92          assertEquals("²", SpamProtector.toEncoded("²"));
93      }
94  
95      @Test
96      public void issue24LowercasingIsDefaultBehaviour() {
97          assertEquals("A", SpamProtector.toPlain(CharacterConverter.CAPITAL_A.getPlain()));
98          assertEquals("a", SpamProtector.toPlain(CharacterConverter.A.getPlain()));
99      }
100 
101 }
