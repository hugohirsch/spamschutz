18  package de.aikiit.spamprotector.util;
19  
20  import org.junit.Test;
21  
22  import java.util.Locale;
23  
24  import static de.aikiit.spamprotector.util.LocalizationHelper.*;
25  
26  import static org.junit.Assert.assertEquals;
27  
28  /**
29   * Testing resource bundling.
30   *
31   * @author hirsch
32   * @version 23.08.11
33   */
34  public class LocalizationHelperTest {
35  
36      /**
37       * Retrieve a i18n-value with parameters.
38       */
39      @Test
40      public final void checkParametrizedValueExtraction() {
41          assertEquals("Erfolg und dann folgt noch die 7", getParameterizedBundleString("spamschutz.test.param",
42                  "Erfolg", 7));
43          // ignore warning, we want to test what happens here! An empty String or null changes the output.
44          assertEquals("{0} und dann folgt noch die {1}", getParameterizedBundleString("spamschutz.test.param",
45                  new Object[]{}));
46      }
47  
48      @Test
49      public final void fallbackLocale() {
50          // WHEN: reset system properties
51          System.setProperty("user.language", "");
52          System.setProperty("user.country", "");
53          setLocale();
54  
55          assertEquals(Locale.GERMANY, getLocale());
56          assertEquals("de", getLanguage());
57      }
58  
59      @Test
60      public final void setLocaleViaSystemProperties() {
61          // WHEN: reset system properties
62          final String french = "fr";
63          System.setProperty("user.language", french);
64          System.setProperty("user.country", "CA");
65          setLocale();
66  
67          assertEquals(Locale.CANADA_FRENCH, getLocale());
68          assertEquals(french, getLanguage());
69      }
70  
71      @Test
72      public final void umlautEncodingWorksCorrectly() {
73          assertEquals("ßäü", getBundleString("spamschutz.test.umlauts"));
74      }
75  
76      @Test
77      public final void unknownKey() {
78          String unknown = "abcde.unknown";
79          assertEquals(unknown, getBundleString(unknown));
80      }
81  }
