27  /**
28   * Application starter class.

30   * @author hirsch

33  @NoArgsConstructor(access = AccessLevel.NONE)
34  public final class AntiSpamApplication {
35  
36      /** Application title to use in Swing window. */
37      private static final String APP_TITLE = "spamschutz - "
38              + "der einfachste Schutz gegen Spider-Spam";
39  
40      /**
41       * Helper that starts and initializes the application itself.
42       */
43      static void initApplication() {
44          JFrame frame = new JFrame();
45          GUI g = new GUI(false);
46          frame.setTitle(APP_TITLE);
47          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
48          frame.setSize(GUI.APP_DIMENSION);
49          frame.add(g);
50          frame.setVisible(true);
51      }
52  
53      /**
54       * Main method that loads the application in a proper size.
55       *
56       * @param args Arguments from console.
57       */
58      public static void main(final String[] args) {
59          initApplication();
60      } // end of main
