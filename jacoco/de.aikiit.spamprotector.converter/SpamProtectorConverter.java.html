    private char plain;
    private String scrambled;


    SpamProtectorConverter(char input, String output) {
        this.plain = input;
        this.scrambled = output;
    }

    @Override
    public String getReplacement() {
        return scrambled;
    }

    @Override
    public char getPlainText() {
        return plain;
    }
